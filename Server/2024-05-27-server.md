## 데이터베이스를 선택 명령
```
# 데이터베이스 선택
use mysql
# 데이터베이스 생성
create database yyydb;
```
![createdb]()

```
# 사용자를 생성(내부에서 접속)
create user 'yyy'@'localhost' identified by '비번';
# 사용자를 생성(외부에서 접속)
create user 'yyy'@'%' identified by '비번';

# 데이터베이스를 사용할 사용자 권한을 부여
grant all privileges on yyydb.* to 'yyy'@'localhost';
grant all privileges on yyydb.* to 'yyy'@'%';

# 권한부여
flush privileges;
exit
```
![권한]()


```
# 내가 만든 데이터베이스 로그인
mysql -u yyy -p 
```
