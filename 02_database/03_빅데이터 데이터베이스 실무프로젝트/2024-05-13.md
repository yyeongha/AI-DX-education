### ubuntu에서 데이터베이스 만들기

su -

use mysql;

create database memdb;

# 사용자
create user 'mem'@'localhost' identified by '@ai1234@';
create user 'mem'@'%' identified by '@ai1234@';

# 권한
grant all privileges on memdb.* to 'mem'@'localhost';
grant all privileges on memdb.* to 'mem'@'%';

# 권한 적용

exit
